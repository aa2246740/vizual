       ```typescript
       // Auto-detect orphaned annotations when sections change (AI revision returns new content)
       useEffect(() => {
         if (!sections || sections.length === 0) return
         // Extract text content from all sections for orphan matching
         const allText = sections
           .map(s => {
             if (typeof s.content === 'string') return s.content
             if (s.data && typeof s.data === 'object') return JSON.stringify(s.data)
             return ''
           })
           .filter(Boolean)
           .join(' ')
         if (allText) {
           onContentRevised(allText)
         }
       }, [sections, onContentRevised])
       ```
